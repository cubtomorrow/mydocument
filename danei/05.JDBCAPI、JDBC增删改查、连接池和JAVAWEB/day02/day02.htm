public class JDBCUtils {

 

   //1,私有化构造函数，不让外界直接new对象了

   private JDBCUtils(){}

  

   static ResourceBundle rb = null;

  

   //a,,在静态代码块中加载一次属性文件即可

   static{

      //a,读取属性文件

       rb = ResourceBundle.getBundle("jdbc");

   }

  

   //2,提供静态的getConnection方法，封装第1,2

   public static Connection getConnection(){

//    封装第1,2

      try {

        

//       1,注册驱动

         Class.forName(rb.getString("driverClass"));

//       2,获取数据库连接

         String url=rb.getString("jdbcUrl");

         String user=rb.getString("user");

         String password=rb.getString("password");

        

         Connection conn =

                DriverManager.getConnection(

                      url, user, password);

     

         return conn;

        

      } catch (Exception e) {

         e.printStackTrace();

      }

     

      return null;

     

   }

  

   //提供静态的close方法，封装第6

   public static void close(

            Connection conn,

                Statement st,

                   ResultSet rs){

      if(conn!=null){//防止空指针

         try {

            conn.close();//释放连接资源

         } catch (SQLException e) {

            e.printStackTrace();

         }

      }

      if(st!=null){//防止空指针

         try {

            st.close();//释放传输器资源

         } catch (SQLException e) {

            e.printStackTrace();

         }

      }

      if(rs!=null){//防止空指针

         try {

            rs.close();//释放结果集资源

         } catch (SQLException e) {

            e.printStackTrace();

         }

      }

     

     

   }

  

  

  

  

  

}

2.3    SQL注入攻击问题

2.3.1  现象

当SQL语句中包含特殊的SQL关键字时，会造成SQL语义的改变。

chenzs'#  ---   #是SQL中的注释内容

chenzs ' or '1=1    ---   or ‘1=1’  SQL中拼接了这个条件，所有的内容都可以被查到

2.3.2  发起的SQL

select * from user  where username='chenzs'#' and password=''

 

 

select * from user  where username='chenzs' or '1=1' and password=''

2.3.3  解决方案

改造login方法，将Statement对象替换成PreparedStatement对象

//利用jdbc技术，查询user表

2.4    PreparedStatement对象的优势

2.4.1  防止SQL注入攻击问题

就是因为把SQL关键字当做普通的文本来处理。

2.4.2  SQL书写简洁

把含有？的一条SQL，这种SQL叫做SQL骨架，先把SQL骨架发送给数据库服务器，参数的值需要手动设置。

ps.setString(x,y);//x―匹配？的位置，x的值要和？的顺序匹配  y―参数的值。

2.4.3  带有缓存的效果，提高程序的执行效率

先把相同的SQL骨架缓存起来，当下次来访问相同骨架的SQL时，直接去缓存中取，而不用数据库了，提高程序的执行效率。

 

3   批处理

3.1    什么是批处理

当需要向数据库中插入大量的SQL时，就用到jdbc批处理的功能。

流程：

将大量的SQL打成多个批次addBatch()，统一发送给数据库服务器executeBatch()。减少数据库的开关次数。

 

3.2    批处理的实现方式

3.2.2  利用Statement对象

3.3    事务优化

为了提高程序效率，还可以进行事务优化。

事务主要包含两个动作：一开启，（执行SQL），一提交。

这时候我们需要做两件事：

1，关闭jdbc的事务管理

2，手动提交事务

//利用新对象完成

   @Test

   public void PSTest(){

      Connection conn =null;

      PreparedStatement ps = null;

     

      try {

         //1,注册驱动  2，获取数据库连接

         conn = JDBCUtils.getConnection();

        

         //a,,,关闭jdbc的事务管理

         conn.setAutoCommit(false);

        

        

         long start = System.currentTimeMillis();

        

         //3,获取传输器

         String sql =

      "insert into account values(null,?,?)";

         ps = conn.prepareStatement(sql);

        

         //设置参数

         for (int i = 0; i < 100; i++) {

            ps.setString(1, i+"PSwanght");

            ps.setString(2, i+"1000");

           

            //把SQL打成一个批次

            ps.addBatch();

         }

        

         //4,执行SQL，统一发送服务器

         int[] rows = ps.executeBatch();

         System.out.println(rows.length);

        

         //b,手动提交事务

         conn.commit();

        

         long end = System.currentTimeMillis();

         System.out.println(end-start);//3196

        

        

      } catch (Exception e) {

      }finally{

         //6,释放资源

         JDBCUtils.close(conn, ps, null);

      }

 

3.4    两种方式的区别

Ø  Statement对象

优点：在一次批处理中，SQL表现形式灵活

缺点：不防止SQL注入，SQL执行效率低

 

Ø  PreparedStatement对象

优点：防止SQL注入，将SQL骨架缓存起来，SQL语句简洁

缺点：在一次批处理中，只能用来处理骨架相同的SQL

3.5    总结

1.  当没有使用批处理时，100个用户需要执行100条SQL，需要打开数据库连接100次，关闭数据库连接100次。

2.  使用完批处理后，可以将一部分SQL打成一个批次，统一把这些SQL发送给服务器，减少数据库的交互。

3.  使用批处理不一定会使程序的执行效率增加，建议配合着事务优化一起使用。

4.  建议不要一次性向批中插入大量的SQL，容易造成内存溢出。

public class MyPool implements DataSource{

  

   //1,创建一个容器List，存放数据库连接

   private static List<Connection> list =

            new LinkedList<Connection>();

  

   //2,创建静态代码块，初始化容器

   static{

      for (int i = 0; i < 3; i++) {

         //通过工具类获取连接，存入list中

         Connection conn = JDBCUtils.getConnection();

         list.add(conn);

      }

   }

  

   //3,提供getConnection方法，对外提供获取连接对象

   @Override

   public Connection getConnection() throws SQLException {

//    list.get(0);//只是拿出来看看

     

      //拿出来看看，从池子里删除掉这个连接

      Connection conn = list.remove(0);

      System.out.println("连接少了一个...");

     

      return conn;

   }

 

   //4,提供returnConn方法，用完连接还回池里

   public void returnConn(Connection conn){

      try {

         if(conn!=null && !conn.isClosed()){

            list.add(conn);

            System.out.println("还回来一个连接...");

         }

      } catch (SQLException e) {

         e.printStackTrace();

      }

   }
